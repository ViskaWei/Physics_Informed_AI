# 特征工程常用操作
import numpy as np

def normalize(X):
    """归一化到 [0, 1]"""
    return (X - X.min()) / (X.max() - X.min() + 1e-10)

def standardize(X):
    """标准化到均值0方差1"""
    return (X - X.mean()) / (X.std() + 1e-10)

def one_hot(y, num_classes):
    """独热编码"""
    return np.eye(num_classes)[y]
