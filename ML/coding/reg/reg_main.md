## 🔧 通用模板

```python
# 线性回归基础
import numpy as np

def linear_regression(X, y):
    # 最小二乘法: w = (X^T X)^{-1} X^T y
    X = np.c_[np.ones(len(X)), X]  # 添加偏置项
    w = np.linalg.lstsq(X, y, rcond=None)[0]
    return w

def predict(X, w):
    X = np.c_[np.ones(len(X)), X]
    return X @ w
```
